//Комплексный калькулятор
document.querySelector('#plus').onclick = () => {
	//Забираем значение из поле ввода
	let aa = document.querySelector('#a1').value;
	let ba = document.querySelector('#b1').value;
	let ab = document.querySelector('#a2').value;
	let bb = document.querySelector('#b2').value;
	//Ссылка на тег результата
	let result = document.querySelector('#calcresult');
	//Делаем проверку на пустые поля
	if (aa === '' || ba === ''||ab === '' || bb === '') {
		return result.innerHTML = 'Заполните все поля';
	}
	else{
		const z1 = math.complex(aa,ba);
		const z2 = math.complex(ab,bb);
		return result.innerHTML = math.add(z1, z2);
	}
}
